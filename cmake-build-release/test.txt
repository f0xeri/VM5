; сложение чисел 2 и 3 (результат в регистре 1) и вывод результата
l 0              ; грузим программу начиная с адреса 0
i 2              ; инт 2 (будет сохранён по адресу 1)
i 3              ; инт 3 (адрес 2)
s 2              ; ip = 2
c 061 0 0 0      ; movrs из адреса 0 в памяти сохраняем в регистр 0
c 061 1 0 1      ; movrs из адреса 1 в памяти сохраняем в регистр 1
c 010 0 1        ; add регистры 0 + 1 (результат будет в 1)
c 062 1 0 5      ; movsr из регистра 1 сохраняем в память по адресу 5
c 002 1 0 5      ; out выводим содержимое памяти по адресу 5
